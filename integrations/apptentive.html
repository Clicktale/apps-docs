Apptentive

Apptentive Integration

When using both Clicktale for Apps and Apptentive, you can now integrate the systems and for each engagement performed with Apptentive, you can see the corresponding Clicktale session.

This can enable you to see the exact experience leading to the engagement inside Apptentive, helping you understand what it is exactly that they need of you right now.

This data integration contains two parts:

  • Connecting Apptentive Conversations to Clicktale sessions
  • Finding Clicktale sessions where Apptentive surveys or app rating engagement was performed

Connecting Apptentive Conversations to Clicktale sessions

In Clicktale for Apps, when a session begins, which may be on first activation (‘splash screen’ like) or when resuming from background after some time of inactivity, a delegate/callback is used to signal that a new session has begun. At that point you could send Apptentive as hidden text messages the URL for that session recording.

The delegate function can be used like this for iOS:

-(void)clicktaleDidSessionURLCreated:(NSString *)sessionLink
{ 
  [[Apptentive shared] sendAttachmentText:[NSString stringWithFormat:@"Clicktale Session URL : %@",sessionLink]];
}

Or with Swift:

func clicktaleDidSessionURLCreated(_ sessionLink: String!) 
{  
  Apptentive.shared.sendAttachment("Clicktale Session URL : " + sessionLink)
}

The callback function can be used like this for Android:

@Override
public void onSessionURLCreated(String sessionLink) {
    Apptentive.sendAttachmentText("Clicktale Session URL : " + sessionLink);
}

It will be seen in Apptentive conversations

The above is just one suggestion on how to send the sessions link, it is also possible to send the attachment at your convenience at any other time during the runtime by calling:

// other code
[[Apptentive shared] sendAttachmentText:[NSString stringWithFormat:@"Clicktale Session URL : %@",[[Clicktale sharedInstance] getSessionLink]]];
// other code

Or with Swift:

// other code
Apptentive.shared.sendAttachment("Clicktale Session URL : " + Clicktale.sharedInstance().getSessionLink())
// other code

On Android:

// other code
Apptentive.sendAttachmentText("Clicktale Session URL : " + Clicktale.getSessionLink())
// other code

Track Apptentive events

In order for Clicktale for Apps to track Apptentive events such as:

  1. Message Center Unread Count Changed
  2. App Rating Flow - User Agreed To Rate The App
  3. Survey Shown
  4. Survey Sent

You can observe Apptentive notifications and track them using the following code snippet:

//
//  ApptentiveTracker.h
// 

#import <Foundation/Foundation.h>

@interface ApptentiveTracker : NSObject
+(void)bindApptentiveNotifications;
@end

//
//  ApptentiveTracker.m
//  

#import "ApptentiveTracker.h"
@import Apptentive;
@import Clicktale;

@implementation ApptentiveTracker

+(void)bindApptentiveNotifications
{
    [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(handleApptentiveMessageCenterUnreadCountChanged:) name:ApptentiveMessageCenterUnreadCountChangedNotification object:nil];
    [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(handleApptentiveAppRatingFlowUserAgreedToRateApp:) name:ApptentiveAppRatingFlowUserAgreedToRateAppNotification object:nil];
    [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(handleApptentiveSurveyShown:) name:ApptentiveSurveyShownNotification object:nil];
    [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(handleApptentiveSurveySent:) name:ApptentiveSurveySentNotification object:nil];
}

+(void)handleApptentiveMessageCenterUnreadCountChanged:(NSNotification *)notification {
    NSString *unreadMessageCount = [NSString stringWithFormat:@"%@", notification.userInfo[@"count"] ?: @0];
    [[Clicktale sharedInstance] trackEvent:@"ApptentiveMessageCenterUnreadCountChanged" value:unreadMessageCount];
}

+(void)handleApptentiveAppRatingFlowUserAgreedToRateApp:(NSNotification *)notification {
    [[Clicktale sharedInstance] trackEvent:@"ApptentiveAppRatingFlowUserAgreedToRateApp"];
}

+(void)handleApptentiveSurveyShown:(NSNotification *)notification {
    NSString *apptentiveSurveyID = notification.userInfo[ApptentiveSurveyIDKey] ?: @"";
    [[Clicktale sharedInstance] trackEvent:@"ApptentiveSurveyShown.SurveyID" value: apptentiveSurveyID];
}

+(void)handleApptentiveSurveySent:(NSNotification *)notification {
    NSString *apptentiveSurveyID = notification.userInfo[ApptentiveSurveyIDKey] ?: @"";
    [[Clicktale sharedInstance] trackEvent:@"ApptentiveSurveySent.SurveyID" value: apptentiveSurveyID];
}

- (void)dealloc
{
    [[NSNotificationCenter defaultCenter] removeObserver:self];
}

@end

Or with Swift:

//
//  ApptentiveTracker.swift
//  

import Apptentive
import Clicktale

class  ApptentiveTracker {

    class func bindApptentiveNotifications()
    {
        NotificationCenter.default.addObserver(self, selector: #selector(handleApptentiveMessageCenterUnreadCountChanged(notification:)), name: NSNotification.Name.ApptentiveMessageCenterUnreadCountChanged, object: nil) 
        NotificationCenter.default.addObserver(self, selector: #selector(handleApptentiveAppRatingFlowUserAgreedToRateApp(notification:)), name: NSNotification.Name.ApptentiveAppRatingFlowUserAgreedToRateApp, object: nil)
        NotificationCenter.default.addObserver(self, selector: #selector(handleApptentiveSurveyShown(notification:)), name: NSNotification.Name.ApptentiveSurveyShown, object: nil)
        NotificationCenter.default.addObserver(self, selector: #selector(handleApptentiveSurveySent(notification:)), name: NSNotification.Name.ApptentiveSurveySent, object: nil)
    }

    @objc class func handleApptentiveMessageCenterUnreadCountChanged(notification : NSNotification) {
        if let count = notification.userInfo?["count"]  {
            Clicktale.sharedInstance().trackEvent("ApptentiveMessageCenterUnreadCountChanged", value: "\(count)")
        }
        else{
             Clicktale.sharedInstance().trackEvent("ApptentiveMessageCenterUnreadCountChanged", value: "0")
        }
    }

    @objc class func handleApptentiveAppRatingFlowUserAgreedToRateApp(notification : NSNotification) {
         Clicktale.sharedInstance().trackEvent("ApptentiveAppRatingFlowUserAgreedToRateApp")
    }

    @objc class func handleApptentiveSurveyShown(notification : NSNotification) {
        if let surveyID = notification.userInfo?[ApptentiveSurveyIDKey]  {
            Clicktale.sharedInstance().trackEvent("ApptentiveSurveyShown.SurveyID", value: "\(surveyID)")
        }
    }

    @objc class func handleApptentiveSurveySent(notification : NSNotification) {
        if let surveyID = notification.userInfo?[ApptentiveSurveyIDKey]  {
            Clicktale.sharedInstance().trackEvent("ApptentiveSurveySent.SurveyID", value: "\(surveyID)")
        }
    }

    deinit {
        NotificationCenter.default.removeObserver(self)
    }
}

Do not forget to call ApptentiveTracker.bindApptentiveNotifications() OR [ApptentiveTracker bindApptentiveNotifications];

Samples for Android:

 unreadMessagesListener = new UnreadMessagesListener() {
      @Override
      public void onUnreadMessageCountChanged(int unreadMessages) {
        Clicktale.logEvent("ApptentiveMessageCenterUnreadCountChanged", unreadMessages);
      }
    };
    Apptentive.addUnreadMessagesListener(unreadMessagesListener);

surveyFinishListener = new OnSurveyFinishedListener() {
      @Override
      public void onSurveyFinished(boolean completed) {
        Clicktale.logEvent("ApptentiveSurveyFinished", String.valueOf(completed));
      }
    };
    Apptentive.setOnSurveyFinishedListener(surveyFinishListener);
